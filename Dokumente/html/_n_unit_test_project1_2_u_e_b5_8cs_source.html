41  private IPlugin SelectPlugin(IPluginManager mgr, IRequest req)
42  {
43  IPlugin plugin = null;
44  float max = 0;
45  foreach (var p in mgr.Plugins)
46  {
47  float canHandle = p.CanHandle(req);
48  if (canHandle > max)
49  {
50  max = canHandle;
51  plugin = p;
52  }
53  }
54 
55  return plugin;
56  }
